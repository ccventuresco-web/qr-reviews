<script>

let rating = 0;
const stars = document.querySelectorAll(".star");

// seleccionar estrellas
stars.forEach(star => {
    star.addEventListener("click", () => {
        rating = star.getAttribute("data-value");

        stars.forEach(s => s.classList.remove("selected"));

        for (let i = 0; i < rating; i++) {
            stars[i].classList.add("selected");
        }
    });
});

function enviar() {

    if (rating == 0) {
        document.getElementById("mensaje").innerText = "Selecciona una puntuación";
        return;
    }

    // 🔥 CAMBIA ESTE LINK
    const googleLink = "https://search.google.com/local/writereview?placeid=ChIJWXtHp-4UWQ0RPO7s5qtVk6Y";

    if (rating >= 4) {
        window.location.href = googleLink;
    } else {
        document.getElementById("mensaje").innerText = "Gracias por tu opinión, la tendremos en cuenta.";
    }
}

</script>
